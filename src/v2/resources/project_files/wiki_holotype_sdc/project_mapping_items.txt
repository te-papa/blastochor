# Mapfile for creating Wikidata items for type specimens - see also SDC file

# Record-level conditions
reject_if do
    fetch("hasRepresentation").absent?
end

# Hardcoded attributes

# Variable attributes
attribute :record_id, path: "id"

attribute :informationWithheld do
    if memo_check(compose("tepapa:collections/fieldcollection/", fetch("evidenceFor.atEvent.id")), "flag_restrictedLocality").present?
	set_value("something something standard statement")
	fallback(fetch("informationWithheld"))
end

attribute :atEvent, path: "evidenceFor.atEvent"

attribute :atLocation do
    get(:atEvent).fetch("atLocation")
end

attribute :recordedBy do
    get(:atEvent).fetch("recordedBy")
end

attribute :collectionEvent do
    get(:atEvent).fetch("id").lookup("fieldcollection")
end

attribute :currentIdentification do
    fetch("identification").first
end

attribute :currentTypeStatus do
    get(:currentIdentification).fetch("typeStatus")
end

attribute :currentTaxon do
    get(:currentIdentification).fetch("id").lookup("taxon")
end

# Record-level enrichments
enrichment :coordinateWorkaround do
    local(:emuRecord, get(:collectionEvent).fetch("id").use_mapping(:emu_events))
	replace(get(:atLocation).fetch("mappingCentroid.lat"), get_local(:emuRecord).fetch("centroidLatOrWhatever"))
	replace(get(:atLocation).fetch("mappingCentroid.lon"), get_local(:emuRecord).fetch("centroidLonOrWhatever"))
	AND SO ON AND SO ON
end
